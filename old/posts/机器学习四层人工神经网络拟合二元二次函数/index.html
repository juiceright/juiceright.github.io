    for i in range(1,epoch+1):
        tp=model(tu)
        loss=lossfn(tp,tc)
        opt.zero_grad()
        loss.backward()
        opt.step()
        if i % 10 == 0 or i == 1:
            print('Epoch %d, Loss %f' % (i, float(loss)))
    return model
